        Soroush Karami is a Ph.D. candidate
        at the <a href="https://www.uic.edu/">University of Illinois at Chicago</a> in the Department of <a href="https://cs.uic.edu/">Computer Science</a>, 
        where he is being advised by <a href="https://www.cs.uic.edu/~polakis/">Jason Polakis</a>.  He

    function createNews(count) {
        fetch('files/news.json')
        .then(response => {
                if (!response.ok) {
                    throw new Error("HTTP error " + response.status);
                }
                return response.json();

        .then(json => {
                createRows(json, count)
        })
    }

    function createRows(news, count){
        while (table.rows.length>0){
            table.deleteRow(0)  
        }
        newsCount = Object.keys(news).length
        if (count ==  -1)
            count = newsCount
        for (i=newsCount; i>newsCount-count; i--){
            item  = i.toString()
            addRow(news[item])
        }
    }

    function addRow(row){
        r = table.insertRow()
        date = r.insertCell()

        desc.innerHTML = row[1]
